            bxsRestart = bxsRestart + 1

// --- FindAndTalkNPC: find cyan NPC, right-click, say dialogue ---
// redCheckX = x-coordinate for IfColor red check (varies per map)
// dialogue = what to say (e.g. "Ask 力戰六大派高手")
// Note: "0000FF" in BGR = red (FF0000 RGB), confirms NPC selected
Sub FindAndTalkNPC(redCheckX, dialogue)
    Dim talkRetry
    talkRetry = 0

    Rem TALK_FIND
    FindColor 448, 205, 1025, 556, "FFFF00", intX, intY
    If intX > 0 And intY > 0 Then
        TracePrint "NPC found at (" & intX & ", " & intY & ")"
        MoveTo intX + 5, intY + 10
        Delay walkDelay
        RightClick 1
        Delay walkDelay
        IfColor redCheckX, 506, "0000FF", 0 Then
            SayString dialogue
            Delay walkDelay
            KeyPress "Enter", 1
            Delay walkDelay
        Else
            talkRetry = talkRetry + 1
            If talkRetry >= 30 Then
                TracePrint "Talk NPC: red check failed after 30 retries"
            Else
                Delay walkDelay
                Goto TALK_FIND
            End If
        End If
    Else
        talkRetry = talkRetry + 1
        If talkRetry >= 30 Then
            TracePrint "Talk NPC: not found after 30 retries"
        Else
            Delay walkDelay
            Goto TALK_FIND
        End If
    End If
End Sub

// --- FindAndKillNPC: find cyan NPC, kill, combat loop with HP check ---
// redCheckX = x-coordinate for IfColor red check
// Sets global killOK = 1 on success, 0 on failure (map empty)
// Note: "0000FF" in BGR = red (FF0000 RGB), confirms NPC selected
Dim killOK
Sub FindAndKillNPC(redCheckX)
    killOK = 0
    Dim fkRetry
    fkRetry = 0

    Rem FK_FIND
    FindColor 448, 205, 1025, 556, "FFFF00", intX, intY
    If intX > 0 And intY > 0 Then
        TracePrint "Kill target found at (" & intX & ", " & intY & ")"
        MoveTo intX + 5, intY + 15
        Delay walkDelay
        RightClick 1
        Delay walkDelay
        IfColor redCheckX, 506, "0000FF", 0 Then
            Call ClearInput()
            SayString "kill"
            Delay walkDelay
            KeyPress "Enter", 1
            Delay walkDelay

            // Combat loop with HP check
            combatStart = now
            Rem FK_COMBAT
            Do
                // Timeout: 120s max combat
                If DateDiff("s", combatStart, now) >= 120 Then
                    TracePrint "Combat timeout (120s), giving up..."
                    Exit Do
                End If

                // HP check: if first HP digit is black (low HP), heal
                IfColor 274, 298, "000000", 0 Then
                    Call ClearInput()
                    SayString "Use 藥酒 7"
                    Delay walkDelay
                    KeyPress "Enter", 1
                    Delay walkDelay
                End If

                // Kill confirmation: yellow text in result area
                FindColor 756, 284, 791, 293, "FFFF00", intX, intY
                If intX > 0 And intY > 0 Then
                    TracePrint "Kill confirmed!"
                    killOK = 1
                    Exit Do
                End If
                Delay walkDelay
            Loop
        Else
            // Red check failed, retry
            fkRetry = fkRetry + 1
            If fkRetry >= 15 Then
                TracePrint "Kill NPC: red check failed after 15 retries, map may be empty"
                killOK = 0
            Else
                Delay walkDelay
                Goto FK_FIND
            End If
        End If
    Else
        fkRetry = fkRetry + 1
        TracePrint "Kill NPC not found, retry #" & fkRetry
        If fkRetry >= 15 Then
            TracePrint "Kill NPC: not found after 5 retries, map empty"
            killOK = 0
        Else
            Delay walkDelay
            Goto FK_FIND
        End If
    End If
End Sub

// --- CheckCamera: detect camera reset and fix with PageDown x20 ---
// When camera is zoomed in (bad), character name green "21FF42" appears at (772,243)
// When camera is correct (zoomed out), that pixel is NOT green
Sub CheckCamera()
    IfColor 772, 243, "21FF42", 0 Then
        TracePrint "Camera reset detected, fixing..."
        HWND = Plugin.Window.Foreground()
        Dim camI
        For camI = 1 To 20
            Call Plugin.Bkgnd.KeyPress(HWND, 34)
            Delay 100
        Next
        Call NineClick()
    End If
End Sub

// RESTART GAME
// ============================================================
Sub RestartGame()
    TracePrint "=== RESTART GAME ==="
    KeyPress "F7", 1

    Rem RG_CLOSE
    MoveTo 1028, 153
    Delay walkDelay
    LeftClick 1
    TracePrint "Game window closed"

    Do
        FindColor 476, 266, 747, 359, "A56D39", intX, intY
        FindColor 476, 266, 747, 359, "A56E3A", intX1, intY1
        If intX > 0 And intY > 0 Then
            MoveTo 595, 404
            Delay walkDelay
            LeftClick 1
            Exit Do
        ElseIf intX1 > 0 And intY1 > 0 Then
            MoveTo 595, 404
            Delay walkDelay
            LeftClick 1
            Exit Do
        End If
        Delay walkDelay
    Loop

    Do
        FindColor 586, 205, 744, 286, "215DE7", intX, intY
        If intX > 0 And intY > 0 Then
            MoveTo intX, intY
            LeftClick 1
            Exit Do
        End If
        Delay walkDelay
    Loop
    Delay gateDelay

    MoveTo 480, 359
    Delay walkDelay
    LeftClick 1
    Delay gateDelay

    Rem RG_LOGIN
    Do
        FindColor 293, 211, 684, 355, "CEFF63", intX, intY
        If intX > 0 And intY > 0 Then
            // Only type login ID if account field is empty (no green text)
            FindColor 750, 530, 810, 550, "00FF00", intX, intY
            If intX <= 0 Or intY <= 0 Then
                SayString "A12345678"
            End If
            MoveTo 796, 679
            Delay walkDelay
            LeftClick 1
            Delay gateDelay
            Exit Do
        End If
        Delay walkDelay
    Loop

    Dim loginTime
    loginTime = now
    Do
        If DateDiff("s", loginTime, now) >= 30 Then
            TracePrint "Login timeout, retrying..."
            Goto RG_CLOSE
            Exit Do
        Else
            FindColor 350, 203, 488, 287, "3979EF", intX, intY
            If intX > 0 And intY > 0 Then
                MoveTo 712, 517
                Delay walkDelay
                LeftClick 1
                Delay walkDelay
                MoveTo 720, 674
                Delay walkDelay
                LeftClick 1
                Exit Do
            End If
        End If
        Delay walkDelay
    Loop

    Do
        IfColor 263, 189, "8CAAC6", 0 Then
            Delay walkDelay
            Call NineClick()
            Exit Do
        End If
        Delay walkDelay
    Loop

    HWND = Plugin.Window.Foreground()
    Dim camI
    For camI = 1 To 20
        Call Plugin.Bkgnd.KeyPress(HWND, 34)
        Delay 100
    Next

    Delay walkDelay
    Call NineClick()
    KeyDown "Ctrl", 1
    Delay walkDelay
    KeyPress "A", 1
    Delay walkDelay
    KeyUp "Ctrl", 1
    Delay walkDelay
    KeyPress "Delete", 1
    Delay walkDelay

    TracePrint "=== RESTART COMPLETE ==="
End Sub
