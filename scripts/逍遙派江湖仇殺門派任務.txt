// RestartGame - F7 stop, close window, wait error dialog, login
Sub RestartGame()

    MoveTo 1028, 153

            Delay walkDelay
            LeftClick 1

        End If
        Delay walkDelay

        FindColor 586, 205, 744, 286, "215DE7", intX, intY

            MoveTo intX, intY
            LeftClick 1

        End If

    MoveTo 480, 359
    Delay walkDelay
    LeftClick 1
    
    Rem RG_LOGIN
    Do
        FindColor 293, 211, 684, 355, "CEFF63", intX, intY
        If intX > 0 And intY > 0 Then
            // Only type login ID if account field is empty (no green text)

            LeftClick 1
            Delay gateDelay
            Exit Do
        End If

    Dim loginTime
    loginTime = now
    Do
        If DateDiff("s", loginTime, now) >= 30 Then
            TracePrint "Login timeout, retrying..."
            Goto RG_CLOSE
            Exit Do
        Else
            FindColor 350, 203, 488, 287, "3979EF", intX, intY
            If intX > 0 And intY > 0 Then
                MoveTo 712, 517
                Delay walkDelay
                LeftClick 1
                Delay walkDelay
                MoveTo 720, 674
                Delay walkDelay
                LeftClick 1
                Exit Do
            End If
        End If
        Delay walkDelay
    Loop
    
    Do
        IfColor 263, 189, "8CAAC6", 0 Then
            Delay walkDelay
            Call NineClick()
            Exit Do
        End If
        Delay walkDelay
    Loop
    
    HWND = Plugin.Window.Foreground()
    Dim camI
    For camI = 1 To 20
        Call Plugin.Bkgnd.KeyPress(HWND, 34)
        Delay 100
    Next

End Sub

// ResetMapTimer — call when entering a new map section to reset the 60s watchdog
Sub ResetMapTimer()
    mapEntryTime = now
    stuckDetected = 0
End Sub

// CheckStuck — if >60s on same map (except dongdajie), restart and set flag
Sub CheckStuck()
    Call DetectMap()
    If mapResult = "dongdajie" Then
        Exit Sub
    End If
    If DateDiff("s", mapEntryTime, now) >= 60 Then
        TracePrint "STUCK >60s on map: " & mapResult & "! Restarting..."
        stuckDetected = 1
        
        // Track consecutive restarts on same map
        If mapResult = lastRestartMap Then
            consecutiveRestarts = consecutiveRestarts + 1
            If consecutiveRestarts >= 2 Then
                TracePrint "Stuck at same map " & consecutiveRestarts & " times, waiting 20s before restart..."
                Delay 20000
            End If
        Else
            lastRestartMap = mapResult
            consecutiveRestarts = 1
        End If
        
        Call RestartGame()
    End If
End Sub

// CheckCameraAngle — if camera at low angle (10AAFF at 708,240-746,250), fix with PageDown x20
Sub CheckCameraAngle()
    Dim hasBlueSky
    hasBlueSky = 0
    FindColor 708, 240, 746, 250, "10AAFF", intX, intY
    If intX > 0 And intY > 0 Then
        hasBlueSky = 1
    End If
    If hasBlueSky = 1 Then
        TracePrint "Camera angle too low (blue sky detected), adjusting..."
        HWND = Plugin.Window.Foreground()
        Dim camFix
        For camFix = 1 To 20
            Call Plugin.Bkgnd.KeyPress(HWND, 34)
            Delay 100
        Next
        TracePrint "Camera angle adjusted"
    End If
End Sub
