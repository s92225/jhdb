// RestartGame — close game, reopen, login, adjust camera
Sub RestartGame()
    TracePrint "=== RESTART GAME ==="
    KeyPress "F7", 1
    
    Rem RG_CLOSE
    MoveTo 1028, 153
    Delay walkDelay
    LeftClick 1
    TracePrint "Game window closed"
    
    Do
        FindColor 476, 266, 747, 359, "A56D39", intX, intY
        FindColor 476, 266, 747, 359, "A56E3A", intX1, intY1
        If intX > 0 And intY > 0 Then
            MoveTo 595, 404
            Delay walkDelay
            LeftClick 1
            Exit Do
        ElseIf intX1 > 0 And intY1 > 0 Then
            MoveTo 595, 404
            Delay walkDelay
            LeftClick 1
            Exit Do
        End If
        Delay walkDelay
    Loop
    
    Do
        FindColor 586, 205, 744, 286, "215DE7", intX, intY
        If intX > 0 And intY > 0 Then
            MoveTo intX, intY
            LeftClick 1
            Exit Do
        End If
        Delay walkDelay
    Loop
    Delay walkDelay
    
    MoveTo 480, 359
    Delay walkDelay
    LeftClick 1
    
    Rem RG_LOGIN
    Do
        FindColor 293, 211, 684, 355, "CEFF63", intX, intY
        If intX > 0 And intY > 0 Then
            // Only type login ID if account field is empty (no green text)
            FindColor 750, 530, 810, 550, "00FF00", intX, intY
            If intX <= 0 Or intY <= 0 Then
                SayString loginID
            End If
            MoveTo 796, 679
            Delay walkDelay
            LeftClick 1
            Delay gateDelay
            Exit Do
        End If
        Delay walkDelay
    Loop
    
    Dim loginTime
    loginTime = now
    Do
        If DateDiff("s", loginTime, now) >= 30 Then
            TracePrint "Login timeout, retrying..."
            Goto RG_CLOSE
            Exit Do
        Else
            FindColor 350, 203, 488, 287, "3979EF", intX, intY
            If intX > 0 And intY > 0 Then
                MoveTo 712, 517
                Delay walkDelay
                LeftClick 1
                Delay walkDelay
                MoveTo 720, 674
                Delay walkDelay
                LeftClick 1
                Exit Do
            End If
        End If
        Delay walkDelay
    Loop
    
    Do
        IfColor 263, 189, "8CAAC6", 0 Then
            Delay walkDelay
            Call NineClick()
            Exit Do
        End If
        Delay walkDelay
    Loop
    
    HWND = Plugin.Window.Foreground()
    Dim camI
    For camI = 1 To 20
        Call Plugin.Bkgnd.KeyPress(HWND, 34)
        Delay 100
    Next
    
    Delay walkDelay
    Call NineClick()
    Call CheckCameraAngle()
    TracePrint "=== RESTART COMPLETE ==="
End Sub

// CheckCameraAngle — if camera at low angle (10AAFF at 708,240-746,250), fix with PageDown x20
Sub CheckCameraAngle()
    Dim hasBlueSky
    hasBlueSky = 0
    FindColor 708, 240, 746, 250, "10AAFF", intX, intY
    If intX > 0 And intY > 0 Then
        hasBlueSky = 1
    End If
    If hasBlueSky = 1 Then
        TracePrint "Camera angle too low (blue sky detected), adjusting..."
        HWND = Plugin.Window.Foreground()
        Dim camFix
        For camFix = 1 To 20
            Call Plugin.Bkgnd.KeyPress(HWND, 34)
            Delay 100
        Next
        TracePrint "Camera angle adjusted"
    End If
End Sub
